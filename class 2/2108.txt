#include <bits/stdc++.h>
using namespace std;

int main() {
	
	int n ; // 1 <= n <= 500,000
	cin >> n;
	
	//input
	vector<int> n_v;
	int in ; // -4000 <= in <= 4000
	for(int i = 0; i < n; i ++)
	{
		cin >> in;
		n_v.push_back(in);
	}
	
	//sort
	sort(n_v.begin(), n_v.end());
	
	vector<int> usedIntPos;
	vector<int> in_p;
	vector<int> in_m;
	
	//**** input vector : n_v
	//산술평균 구하기
	float ans1 = 0;
	for(int i = 0; i < n; i++)
	{
		ans1 += n_v[i];
	}
	if( (ans1 / n) - ((int) ans1 / n) > 0.5) // 반올림
	{
		cout << (int) (ans1 / n) + 1 << "\n";
	}
	else
	{
		cout << (int) (ans1 / n) << "\n";
	}
	
	//중앙값
	cout << n_v[n_v.size() / 2] << "\n";
	
	//최빈값, 여러개면 두번째로 작은 값 출력
	
	
	//넷째줄에는 범위를 출력
	if(n_v[0] > 0)
	{
		if(n_v[n - 1] > 0)
		{
			cout << n_v[0] + n_v[n-1] << "\n";
		}
		else
		{
			cout << n_v[0] + n_v[n-1] * -1 << "\n";
		}
	}
	else
	{
		if(n_v[n - 1] > 0)
		{
			cout << n_v[0] * -1 + n_v[n-1] << "\n";
		}
		else
		{
			cout << n_v[0] * -1 + n_v[n-1] * -1 << "\n";
		}
	}
	return 0;
}
