#include <bits/stdc++.h>
using namespace std;

int main()
{
	int col;
	int row;
	
	/*
	0	= 0, 1, 2, 3, ... col;
	1	= 0, 1, 2, 3, ... col;
	2
	3
	...
	row
	*/
	cin >> row >> col;
	//chess input
	vector<vector<char>> chess;
	for(int i = 0; i < row; i++)
	{
		vector<char>* dn_v = new vector<char>(col, 0);	
		chess.push_back(*dn_v);
		dn_v = nullptr;
		delete dn_v;
	}
	
	char input;
	for(int i = 0; i < row; i++)
	{
		for(int j = 0; j < col; j++)
		{
			cin >> input;
			chess[i].push_back(input);
		}
	}
	
	for(int i = 0; i< row; i++)
	{
		for(int j = 0; j < col; j++)
		{
			cout << chess[i][j];
		}
		cout << endl;
	}
	
	return 0;
}
